 rails项目中 `Gemfile`文件设置笔记:

```ruby
 gem 'pg'
 gem 'rails_12factor', group: :production

 group :development, :test do
   gem 'sqlite3'
   # ....
 end
 ```

 `database.yml`:

 ```ruby
 production:
   adapter: postgresql
   encoding: unicode
 ```
